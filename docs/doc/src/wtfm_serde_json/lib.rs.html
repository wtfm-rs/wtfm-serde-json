28//! ## It works!
29//! ```
30//! use serde_json::{Result, Value};
31//!
32//! fn main() -> Result<()> {
33//!    // Some JSON input data as a &str. Maybe this comes from the user.
34//!    let data = r#"
35//!        {
36//!            "name": "John Doe",
37//!            "age": 43,
38//!            "phones": [
39//!                "+44 1234567",
40//!                "+44 2345678"
41//!            ]
42//!        }"#;
43//!
44//!    // Parse the string of data into serde_json::Value.
45//!    let v: Value = serde_json::from_str(data)?;
46//!
47//!    // Access parts of the data by indexing with square brackets.
48//!    debug_assert_eq!(
49//!        format!("Please call {} at the number {}", v["name"], v["phones"][0]),
50//!        "Please call \"John Doe\" at the number \"+44 1234567\""
51//!    );
52//!
53//!    Ok(())
54//! }
55//! ```
